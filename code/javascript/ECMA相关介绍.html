# Symbol

新的原始数据类型，表示独一无二的值，它是JavaScript的第七种数据类型，是一种类似于字符串的数据类型。

  1. Symbol是唯一的，用来解决命名冲突的问题；
  2. Symbol值不能与其他数据进行运算；
  3. Symbol定义的对象属性不能使用for..in循环遍历，但是可以使用Reflect.ownKeys来获取对象所有链明。
// 创建Symbol
let s = Symbol();
console.log(s, typeof s); // Symbol() "symbol"

let s2 = Symbol('描述'); // 只作为一个标志
let s3 = Symbol('描述');
console.log(s2 === s3); // false

const xiyou = ["A","B","C","D"]

for(let item in xiyou){
	console.log(item) // A B C D
}

let iterator = xiyou[Symbol.iterator]();
console.log(iterator.next()); //{value: "A", done: flase}
console.log(iterator.next()); //{value: "B", done: flase}
console.log(iterator.next()); //{value: "C", done: flase}
console.log(iterator.next()); //{value: "D", done: flase}
console.log(iterator.next()); //{value: undefined, done: true}

    setTimeout(function(){
        let data = "UserData";
        resolve(data); //对象状态变为成功
        // reject();如果调用reject，promise对象的状态变为失败。
    }, 1000);
});

p.then(function(value){
    console.log(value);
},function(reason){

# 生成随机字符串

export function uuid(length, chars){
    chars = chars || '0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
    length = length || 8
    var result = ""
    for(var i = length; i > 0; --i){
        result += char[Math.floor(Math.random() * chars.length)]
    }
    return result;
}

export function uniqueArray(arr){
    if(!Array.isArray(arr)){
        throw new Error('is not Array')
    }
    if(arr.length == 1){
        return arr;
    }
    return [...new Set(arr)]

export function arrayDifference(arr1, arr2){
    if(!Array.isArray(arr1) || !Array.isArray(arr2)){
        throw new Error('is not Array')
    }
    return [...new Set([...arr1].filter(x => arr2.has(x)))]

# 对象转换为FormData对象

export function getFormData(obj) {
    const formData = new FormData();
    Object.keys(obj).forEach(key =>{
        const value = obj[key];
        if (Array.isArray(value)) {
            value.forEach((subValue, i) => {
                formData.append(key + `[${i}]`, subValue);
            })
        }else {
            formData.append(key, obj[key])
        }
    })
    return formData;
}

// keys: 返回一个由一个给定对象的自身可枚举属性组成的数组，数组中属性名的排列书序和正常遍历该对象时返回的顺序一致
// append: 会添加一个新值到formData对象内的一个已存在的键中，如果键不存在则会添加该键

function download(link, name){
    if (!name) {

    let eleLink = document.createElement('a')
    eleLink.download = name;
    eleLink.style.display = 'none';

# 节流

/**
 * 节流，多次触发，间隔时间段执行
 * @param {Function} func
 * @param {Int} wait
 * @param {Object} options
 */
export function throttle(func, wait = 500, options) {
  var timeout, context, args
  //   以前的
  var previous = 0
  //   如果options不存在，则给上默认值
  if (!options) {
    options = { leading: false, trailing: true }
  }

  var later = function () {
    //  如果不是立即执行，则为0，立即执行则获取当前的时间戳
    previous = options.leading === false ? 0 : new Date().getTime()
    timeout = null
    // 调用一个具有给定this值的函数，以及一个数组（或类数组对象）的形式提供的参数
    func.apply(context, args)
    if (!timeout) {
      // 连续赋值
      context = args = null
    }
  }

  var throttled = function () {
    //   获取当前时间戳
    var now = new Date().getTime()
    // 跟17行代码一样的效果
    if (!previous && options.leading === false) {
      previous = now
    }
    // remaining:剩下的
    var remaining = wait - (now - previous)
    context = this
    // argument: 对应于传递给函数的参数的类数组对象，实际表现为将函数的多个参数封装到一个数组中
    args = arguments
    // 没有等待时间或者等待时间超过一轮
    if (remaining <= 0 || remaining > wait) {
      // 如果定时器存在，则清除定时器
      if (timeout) {
        clearTimeout(timeout)
        timeout = null
      }
      previous = now
      func.apply(context, args)
      if (!timeout) {
        context = args = null
      } else {
        //   延时执行
        timeout = setTimeout(later, remaining)
      }
    }
  }

  return throttled
}

/* 
options:{leading, trailing}
leading: 函数在每个等待时延的开始被调用，默认值false
trailing： 函数在每个等待时延的结束被调用，默认值true
根据不同的值设置不同的效果：
false true: 默认情况，时延结束后才会调用
true true: 开始以及结束时都会被调用
true false: 只有在开始是会被调用
*/
